func threeSum(nums []int) [][]int {

    sort.Ints(nums)

    n := len(nums)

    ans := make([][]int, 0)

    for i, val := range nums {

        if i > 0 && nums[i-1] == val {

            continue

        }

        if val>0 {

            break

        }

        x, y := i+1, n-1

        for x < y {

            sum := val + nums[x] + nums[y]

            if sum == 0 {

                ans = append(ans, []int{val, nums[x], nums[y]})

                //去重

                for x < y && nums[x] == nums[x+1] {

                    x++

                }

                for x < y && nums[y] == nums[y-1] {

                    y--

                }

                x++

                y--

            } else if sum > 0 {

                y--

            } else {

                x++

            }

        }

    }

    return ans

}
