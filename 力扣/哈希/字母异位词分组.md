- 属于同一组的词排序后是同一个字符串,那么就用map用sorted作为key来维持同一组字符串
```go
func groupAnagrams(strs []string) [][]string {

    m:=make(map[string][]string,0)

    for _,s:=range strs {

        ch:=[]byte(s)

        sort.Slice(ch,func(i,j int) bool {

            return ch[i]<ch[j]

        })

        sorted:=string(ch)

        m[sorted]=append(m[sorted],s)

    }

    ans:=make([][]string,0)

    for _,v:=range m {

        ans=append(ans,v)

    }

    return ans

}
```
